def make(path)
  match_maker do
    message { "to [create|not create] the #{path}" }
    before do
      @existed = File.exists? path
    end
    test do
      !@existed && File.exists?(path)
    end
  end
end

def show_help
  match_maker do
    message { "to [show|not show] the default help text, got #{lines.first.inspect}" }
    test { (%rhelp.*code gen/ =~ lines[1]) }
  end
end
