def stamp(opt={})
  opt[:binding] = binding
  @gg.stamp_generator 'mirror', filename, opt
  @gg.stamp_module 'mirror-abstract', "abstract_#{filename}", opt
  @gg.stamp_module 'mirror-impl', filename, opt
