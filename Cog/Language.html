def default_lit_for(name)
  case name
  when :boolean
    to_boolean false
  when :integer
    to_integer 0
  when :long
    to_long 0
  when :float
    to_lit 0.0
  when :double
    to_double 0.0
  when :char
    to_char ''
  when :string
    to_string ''
  when :null
    to_null nil
  end
end

def to_ident(name)
  if @reserved.member? name.to_s
    "#{name}_"
  else
    name.to_s
  end
end

def to_prim(name)
  ident = @prim_ident[name.to_sym]
  raise Errors::PrimitiveNotSupported.new :type => name, :language => @name unless ident
  ident
end

def to_s(w=nil)
  w ||= @name.length
  "#{@name.ljust w} -> #{@extensions.collect {|x| x.to_s}.sort.join ', '}"
end
