def comment(text)
  if text =~ /\n/
    multi_line_comment text
  else
    one_line_comment text
  end
end

def to_s(w=nil)
  w ||= @name.length
  "#{@name.ljust w} -> #{@extensions.collect {|x| x.to_s}.sort.join ', '}"
end
