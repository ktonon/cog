def initialize(details={})
  @details = if details.is_a? Hash
    details.to_a.collect do |key, value|
      "#{key} => #{value.inspect}"
    end.sort
  else
    [details]
  end
end

def message
  w = custom_message || self.class.name.underscore.split('/').last.gsub('_', ' ')
  w += " (#{@details.join ', '})" unless @details.empty?
  w
end
