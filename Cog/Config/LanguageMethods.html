- (Object) activate_language(opt = {}) { ... }

Activate a given language within the scope of the provided block. Either provide :id or :ext but not both. If the extension does not match any of the supported languages, the #active_language will not change, but the block will still be called.

Parameters:

  • opt (Hash) (defaults to: {}) —

    a customizable set of options

Options Hash (opt):

  • :id (:String) — default: nil —

    the lanuage identifier. Type cog language list to see the possible values

  • :ext (:String) — default: nil —

    a file extension which will map to a language identifier. Type cog language map to see mapped extensions

Yields:

  • within this block the #active_language will be set to the desired value

Returns:

  • (Object) —

    the value returned by the block

def activate_language(opt={}, &block)
  throw :ActivateLanguageRequiresABlock if block.nil?
  lang_id = if opt[:ext]
    ext = opt[:ext].to_s
    ext = ext.slice(1..-1) if ext.start_with?('.')
    @language_extension_map[ext.downcase.to_sym] unless ext.empty?
  else
    opt[:id]
  end
  if lang_id
    @active_languages << Languages.get_language(lang_id)
    r = block.call
    @active_languages.pop
    r
  else
    block.call
  end
end

def language_for(path)
  ext = File.extname(path.to_s)
  ext = path.to_s if ext.empty?
  ext = ext.downcase
  ext = ext.slice(1..-1) if ext.start_with? '.'
  lang_id = @language_extension_map[ext.to_sym]
  Languages.get_language lang_id unless lang_id.nil?
end

def language_summary
  summary = {}
  @language_extension_map.each_pair do |ext, lang_id|
    lang_id = Languages::ALIAS[lang_id] || lang_id
    summary[lang_id] ||= LangInfo.new(lang_id, Languages::REV_ALIAS[lang_id] || [])
    summary[lang_id].extensions << ext
  end
  summary.values.sort
end
